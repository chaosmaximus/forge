Forge has been deployed!

1. Get the application URL:
{{- if eq .Values.service.type "ClusterIP" }}
  kubectl port-forward svc/{{ include "forge.fullname" . }} {{ .Values.service.port }}:{{ .Values.service.port }}
  echo "Forge is available at http://localhost:{{ .Values.service.port }}"
{{- else if eq .Values.service.type "NodePort" }}
  export NODE_PORT=$(kubectl get svc {{ include "forge.fullname" . }} -o jsonpath="{.spec.ports[0].nodePort}")
  export NODE_IP=$(kubectl get nodes -o jsonpath="{.items[0].status.addresses[0].address}")
  echo "Forge is available at http://$NODE_IP:$NODE_PORT"
{{- else if eq .Values.service.type "LoadBalancer" }}
  echo "Waiting for LoadBalancer IP..."
  kubectl get svc {{ include "forge.fullname" . }} -w
{{- end }}

  curl http://localhost:{{ .Values.service.port }}/healthz
