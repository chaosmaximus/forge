```python
"""LadybugDB connection manager with write serialization."""
import asyncio
import os
from pathlib import Path

import real_ladybug as lb


class GraphDB:
    """Manages a single LadybugDB instance with write lock."""

    def __init__(self, db_path: str | Path) -> None:
        self._path = Path(db_path)
        self._path.parent.mkdir(parents=True, exist_ok=True)
        self._db = lb.Database(str(self._path))
        self._conn = lb.Connection(self._db)
        self._write_lock = asyncio.Lock()

    @property
    def conn(self) -> lb.Connection:
        return self._conn

    async def execute(self, query: str, parameters: dict | None = None) -> lb.QueryResult:
        """Execute a read query (no lock needed)."""
        return self._conn.execute(query, parameters=parameters or {})

    async def write(self, query: str, parameters: dict | None = None) -> lb.QueryResult:
        """Execute a write query (serialized via lock, 5s timeout)."""
        try:
            async with asyncio.timeout(5):
                async with self._write_lock:
                    return self._conn.execute(query, parameters=parameters or {})
        except asyncio.TimeoutError:
            raise TimeoutError("Write lock timeout (5s) — another write in progress")

    def close(self) -> None:
        self._db.close()
```

```python
import time
from dataclasses import dataclass, field


@dataclass
class ToolMeta:
    """Tracks metadata for a single tool invocation."""
    start_time: float = field(default_factory=time.monotonic)
    tokens_input: int = 0
    tokens_output: int = 0
    llm_calls: int = 0
    path: str = "deterministic"

    def finish(self) -> dict:
        return {
            "tokens_input": self.tokens_input,
            "tokens_output": self.tokens_output,
            "llm_calls": self.llm_calls,
            "duration_ms": round((time.monotonic() - self.start_time) * 1000),
            "path": self.path,
        }
```

```python
import real_ladybug as lb

SCHEMA_VERSION = 1

# Node table DDL
NODE_TABLES = [
    """CREATE NODE TABLE IF NOT EXISTS Decision(
        id STRING PRIMARY KEY,
        title STRING,
        rationale STRING,
        status STRING DEFAULT 'active',
        created_at TIMESTAMP DEFAULT timestamp(),
        updated_at TIMESTAMP DEFAULT timestamp(),
        valid_at TIMESTAMP,
        invalid_at TIMESTAMP,

    )""",

    """CREATE NODE TABLE IF NOT EXISTS Skill(

EDGE_TABLES = [
    "CREATE REL TABLE IF NOT EXISTS SUPERSEDES(FROM Decision TO Decision, valid_at TIMESTAMP, invalid_at TIMESTAMP, reason STRING)",
    "CREATE REL TABLE IF NOT EXISTS MOTIVATED_BY(FROM Decision TO Lesson, strength DOUBLE DEFAULT 1.0)",
    "CREATE REL TABLE IF NOT EXISTS FOLLOWS(FROM Decision TO Pattern, confidence DOUBLE DEFAULT 1.0)",
    "CREATE REL TABLE IF NOT EXISTS CONTRADICTS(FROM Decision TO Decision, detected_at TIMESTAMP, resolved BOOLEAN DEFAULT false)",
    "CREATE REL TABLE IF NOT EXISTS AFFECTS(FROM Decision TO Function, impact STRING DEFAULT 'medium')",
    "CREATE REL TABLE IF NOT EXISTS AFFECTS_CLASS(FROM Decision TO Class, impact STRING DEFAULT 'medium')",
    "CREATE REL TABLE IF NOT EXISTS AFFECTS_FILE(FROM Decision TO File, impact STRING DEFAULT 'medium')",
    "CREATE REL TABLE IF NOT EXISTS LEARNED_IN(FROM Lesson TO Session)",
    "CREATE REL TABLE IF NOT EXISTS DECIDED_IN(FROM Decision TO Session)",
    "CREATE REL TABLE IF NOT EXISTS EVOLVED_FROM(FROM Skill TO Skill, evolution_type STRING, diff_ref STRING)",
    "CREATE REL TABLE IF NOT EXISTS APPLIED_IN(FROM Skill TO Session, outcome STRING, tokens_input INT64 DEFAULT 0, tokens_output INT64 DEFAULT 0, llm_calls INT64 DEFAULT 0)",
    "CREATE REL TABLE IF NOT EXISTS LOCATED_IN(FROM Secret TO File)",
]


def create_schema(conn: lb.Connection) -> None:
    """Create all memory node and edge tables. Idempotent."""
    for ddl in NODE_TABLES:
        conn.execute(ddl)
    for ddl in EDGE_TABLES:
        conn.execute(ddl)

    # Version tracking table
    conn.execute("""
        CREATE NODE TABLE IF NOT EXISTS _forge_meta(
            key STRING PRIMARY KEY,
            value STRING
        )
    """)
    # Set version if not exists
    result = conn.execute(
        "MATCH (m:_forge_meta {key: 'schema_version'}) RETURN m.value"
    )
    if result.get_as_pl().is_empty():
        conn.execute(
            "CREATE (m:_forge_meta {key: 'schema_version', value: $v})",
            parameters={"v": str(SCHEMA_VERSION)},
        )
```

```python
"""Schema migration framework — versioned, reversible, preflight-checked."""
import shutil
from pathlib import Path

import real_ladybug as lb


def get_schema_version(conn: lb.Connection) -> int:
    """Get current schema version from _forge_meta table."""
    try:
        result = conn.execute(
            "MATCH (m:_forge_meta {key: 'schema_version'}) RETURN m.value AS v"
        )
        rows = result.get_as_pl()
        if rows.is_empty():
            return 0
        return int(rows["v"][0])
    except Exception:
        return 0


def set_schema_version(conn: lb.Connection, version: int) -> None:
    """Update schema version in _forge_meta."""
    conn.execute(
        "MATCH (m:_forge_meta {key: 'schema_version'}) SET m.value = $v",
        parameters={"v": str(version)},
    )


def backup_db(db_path: Path) -> Path:
    """Create a backup of the database before migration."""
    backup_path = db_path.with_suffix(".lbdb.bak")
    if db_path.exists():
        shutil.copytree(str(db_path), str(backup_path), dirs_exist_ok=True)
    return backup_path


def restore_from_backup(db_path: Path, backup_path: Path) -> None:
    """Restore database from backup on migration failure."""
    if backup_path.exists():
        shutil.rmtree(str(db_path), ignore_errors=True)
        shutil.copytree(str(backup_path), str(db_path))


# Migration registry: version -> (up_fn, down_fn)
MIGRATIONS: dict[int, tuple] = {}
# v1 is the initial schema — created by schema.create_schema()
# Future migrations go here as:
# MIGRATIONS[2] = (migrate_v1_to_v2, migrate_v2_to_v1)
```

```python
"""Trust level classification and content sanitization."""
import re


# Patterns to strip from memory content before context injection
_DANGEROUS_PATTERNS = [
    re.compile(r"<tool_use>.*?</tool_use>", re.DOTALL),
    re.compile(r"<tool_result>.*?</tool_result>", re.DOTALL),
    re.compile(r"`[^`]*(?:rm\s+-rf|sudo|chmod|chown|curl\s+\||\beval\b)[^`]*`"),
    re.compile(r"https?://\S+"),
    re.compile(r"\$\(.*?\)"),
]


def sanitize_for_context(text: str) -> str:
    """Strip dangerous patterns from memory content before injection into LLM context."""
    result = text
    for pattern in _DANGEROUS_PATTERNS:
        result = pattern.sub("[REDACTED]", result)
    return result.strip()


def trust_filter(trust: str = "high") -> str:
    """Return Cypher WHERE clause for trust level filtering."""
    if trust == "high":
        return "trust_level = 'user'"
    if trust == "agent":
        return "trust_level = 'agent'"
    return ""  # "all" — no filter
```

```python
"""Per-agent ACL enforcement for MCP tools."""
from enum import Enum


class AgentRole(Enum):
    LEAD = "lead"           # Full access — human session
    PLANNER = "planner"     # Read-only
    GENERATOR = "generator" # Read-only
    EVALUATOR = "evaluator" # Read-only
    EVOLUTION = "evolution"  # Write to Skill/Pattern only
    READONLY = "readonly"   # Default for unknown agents


# Tools each role can access
_READ_TOOLS = frozenset({
    "forge_recall", "forge_decisions", "forge_patterns", "forge_timeline",
    "forge_usage", "forge_health", "forge_scan",
    "axon_query", "axon_context", "axon_impact", "axon_dead_code",
    "axon_detect_changes", "axon_cypher", "axon_communities", "axon_coupling",
    "axon_call_path", "axon_cycles", "axon_file_context", "axon_test_impact",
    "axon_review_risk", "axon_explain", "axon_list_repos",
})

_WRITE_TOOLS = frozenset({
    "forge_remember", "forge_link", "forge_forget",
})

_ALL_TOOLS = _READ_TOOLS | _WRITE_TOOLS

_ROLE_PERMISSIONS: dict[AgentRole, frozenset[str]] = {
    AgentRole.LEAD: _ALL_TOOLS,
    AgentRole.PLANNER: _READ_TOOLS,
    AgentRole.GENERATOR: _READ_TOOLS,
    AgentRole.EVALUATOR: _READ_TOOLS,
    AgentRole.EVOLUTION: _READ_TOOLS | {"forge_remember", "forge_link"},
    AgentRole.READONLY: _READ_TOOLS,
}

_AGENT_ID_TO_ROLE: dict[str, AgentRole] = {
    "forge-planner": AgentRole.PLANNER,
    "forge-generator": AgentRole.GENERATOR,
    "forge-evaluator": AgentRole.EVALUATOR,
    "forge-evolution": AgentRole.EVOLUTION,
}


def get_role(agent_id: str | None) -> AgentRole:
    """Map agent_id to role. None/missing → LEAD (human session)."""
    if agent_id is None:
        return AgentRole.LEAD
    return _AGENT_ID_TO_ROLE.get(agent_id, AgentRole.READONLY)


def check_access(agent_id: str | None, tool: str) -> bool:
    """Check if the given agent can call the given tool."""
    role = get_role(agent_id)
    allowed = _ROLE_PERMISSIONS.get(role, _READ_TOOLS)
    return tool in allowed
```

```python
"""Forge memory MCP tools — 9 tools for the unified knowledge graph."""
import json
import uuid
from datetime import datetime, timezone

from forge_graph.auth import check_access
from forge_graph.db import GraphDB
from forge_graph.meta import ToolMeta
from forge_graph.server import mcp, get_db

# --- Allowed types and edge types ---

ALLOWED_TYPES = frozenset({"decision", "pattern", "lesson", "preference"})
ALLOWED_EDGE_TYPES = frozenset({
    "SUPERSEDES", "MOTIVATED_BY", "FOLLOWS", "CONTRADICTS",
    "AFFECTS", "AFFECTS_CLASS", "AFFECTS_FILE",
    "LEARNED_IN", "DECIDED_IN", "EVOLVED_FROM", "APPLIED_IN", "LOCATED_IN",
})


def _gen_id(type_name: str) -> str:
    return f"{type_name}-{uuid.uuid4().hex[:12]}"


def _remember_impl(
    db: GraphDB,
    type: str,
    structured: dict | None = None,
    content: str | None = None,
    links_to: list[str] | None = None,
    agent_id: str | None = None,
) -> str:
    """Core implementation for forge_remember (testable without MCP)."""
    meta = ToolMeta()

    if not check_access(agent_id, "forge_remember"):
        raise PermissionError(f"Agent '{agent_id}' cannot write memory")

    if type not in ALLOWED_TYPES:
        raise ValueError(f"Unknown type '{type}'. Allowed: {sorted(ALLOWED_TYPES)}")

    if structured is None and content is None:
        raise ValueError("Either 'structured' or 'content' must be provided")

    # Structured path — deterministic, zero LLM
    if structured is not None:
        node_id = _gen_id(type)
        now = datetime.now(timezone.utc).isoformat()

        if type == "decision":
            db.conn.execute(
                """CREATE (d:Decision {
                    id: $id, title: $title, rationale: $rationale,
                    status: 'active', created_at: $now, updated_at: $now,
                    confidence: $conf, trust_level: $trust
                })""",
                parameters={
                    "id": node_id,
                    "title": structured.get("title", ""),
                    "rationale": structured.get("rationale", ""),
                    "now": now,
                    "conf": structured.get("confidence", 1.0),
                    "trust": "user",
                },
            )
        elif type == "pattern":
            db.conn.execute(
                """CREATE (p:Pattern {
                    id: $id, name: $name, description: $desc,
                    domain: $domain, confidence: $conf, generation: 0,
                    created_at: $now, updated_at: $now
                })""",
                parameters={
                    "id": node_id,
                    "name": structured.get("name", ""),
                    "desc": structured.get("description", ""),
                    "domain": structured.get("domain", "general"),
                    "now": now,
                    "conf": structured.get("confidence", 0.5),
                },
            )
        elif type == "lesson":
            db.conn.execute(
                """CREATE (l:Lesson {
                    id: $id, insight: $insight, context: $ctx,
                    severity: $sev, created_at: $now, updated_at: $now
                })""",
                parameters={
                    "id": node_id,
                    "insight": structured.get("insight", ""),
                    "ctx": structured.get("context", ""),
                    "sev": structured.get("severity", "info"),
                    "now": now,
                },
            )
        elif type == "preference":
            db.conn.execute(
                """CREATE (p:Preference {
                    id: $id, key: $key, value: $val,
                    scope: $scope, confidence: $conf,
                    created_at: $now, updated_at: $now
                })""",
                parameters={
                    "id": node_id,
                    "key": structured.get("key", ""),
                    "val": structured.get("value", ""),
                    "scope": structured.get("scope", "project"),
                    "now": now,
                    "conf": structured.get("confidence", 1.0),
                },
            )

        return json.dumps({"node_id": node_id, "_meta": meta.finish()})

    # Natural language path — requires LLM (not implemented in v0.2.0 initial)
    # For now, raise a helpful error
    raise NotImplementedError(
        "Natural language forge_remember requires LLM extraction. "
        "Use structured input for v0.2.0."
    )

    Use 'structured' for deterministic storage (zero LLM tokens).
    Use 'content' for natural language extraction (1 LLM call).
    """
    db = get_db()
    return _remember_impl(db, type, structured, content, links_to, agent_id)
```

```python
def _recall_impl(
    db: GraphDB,
    query: str,
    type: str | None = None,
    timeframe: str | None = None,
    limit: int = 20,
    include_historical: bool = False,
    agent_id: str | None = None,
) -> str:
    """Core implementation for forge_recall — FTS-first search cascade."""
    meta = ToolMeta()
    results = []

    # Build type-specific queries
    node_types = [type] if type else ["decision", "pattern", "lesson", "preference"]
    temporal = "AND n.invalid_at IS NULL" if not include_historical else ""

    for nt in node_types:
        if nt == "decision":
            cypher = f"""
                MATCH (n:Decision)
                WHERE (n.title CONTAINS $q OR n.rationale CONTAINS $q) {temporal}
                RETURN n.id AS id, n.title AS title, n.rationale AS rationale,
                       n.status AS status, n.confidence AS confidence,
                       'decision' AS _type
                LIMIT $lim
            """
        elif nt == "pattern":
            cypher = f"""
                MATCH (n:Pattern)
                WHERE (n.name CONTAINS $q OR n.description CONTAINS $q) {temporal}
                RETURN n.id AS id, n.name AS title, n.description AS rationale,
                       n.domain AS status, n.confidence AS confidence,
                       'pattern' AS _type
                LIMIT $lim
            """
        elif nt == "lesson":
            cypher = f"""
                MATCH (n:Lesson)
                WHERE (n.insight CONTAINS $q OR n.context CONTAINS $q) {temporal}
                RETURN n.id AS id, n.insight AS title, n.context AS rationale,
                       n.severity AS status, 1.0 AS confidence,
                       'lesson' AS _type
                LIMIT $lim
            """
        elif nt == "preference":
            cypher = f"""
                MATCH (n:Preference)
                WHERE (n.key CONTAINS $q OR n.value CONTAINS $q) {temporal}
                RETURN n.id AS id, n.key AS title, n.value AS rationale,
                       n.scope AS status, n.confidence AS confidence,
                       'preference' AS _type
                LIMIT $lim
            """
        else:
            continue

        result = db.conn.execute(cypher, parameters={"q": query, "lim": limit})
        rows = result.get_as_pl()
        for i in range(len(rows)):
            results.append({
                "id": rows["id"][i],
                "title": rows["title"][i],
                "rationale": rows["rationale"][i],
                "status": rows["status"][i],
                "confidence": rows["confidence"][i],
                "_type": rows["_type"][i],
            })

    return json.dumps({"results": results[:limit], "_meta": meta.finish()})


@mcp.tool()
async def forge_recall(
    query: str,
    type: str | None = None,
    timeframe: str | None = None,
    limit: int = 20,
    include_historical: bool = False,
    agent_id: str | None = None,
) -> str:
    """Search the knowledge graph by concept, type, or timeframe.

    FTS-first cascade: text search → vector similarity → LLM rerank (only if >10 results).
    """
    db = get_db()
    return _recall_impl(db, query, type, timeframe, limit, include_historical, agent_id)
```

```python
            f"Allowed: {sorted(ALLOWED_EDGE_TYPES)}"
        )

    if properties:
        prop_parts = []
        for k, v in properties.items():
            param_key = f"p_{k}"
            prop_parts.append(f"{k}: ${param_key}")

        props = ", " + ", ".join(prop_parts) if prop_parts else ""

        MATCH (a:{from_label} {{id: $fid}}), (b:{to_label} {{id: $tid}})
        CREATE (a)-[:{edge_type} {{{props.lstrip(', ')}}}]->(b)
    """
    if not props:
        cypher = f"""
            MATCH (a:{from_label} {{id: $fid}}), (b:{to_label} {{id: $tid}})
            CREATE (a)-[:{edge_type}]->(b)
        """

    db.conn.execute(cypher, parameters=params)

    return json.dumps({"status": "linked", "_meta": meta.finish()})
```

```python
"""Secret scanner — regex + entropy-based detection."""
import hashlib
import math
from dataclasses import dataclass

from forge_graph.security.rules import RULES


@dataclass
class SecretFinding:
    rule_id: str
    provider: str
    type: str
    file_path: str
    line_number: int
    fingerprint: str  # SHA256 of first4+last4 chars — NEVER the full secret
    risk_level: str
    description: str


def _shannon_entropy(s: str) -> float:
    """Calculate Shannon entropy of a string."""
    if not s:
        return 0.0
    freq: dict[str, int] = {}
    for c in s:
        freq[c] = freq.get(c, 0) + 1
    length = len(s)
    return -sum((count / length) * math.log2(count / length) for count in freq.values())


def _fingerprint(value: str) -> str:
    """Create a non-reversible fingerprint: SHA256 of first4 + last4 chars."""
    if len(value) < 8:
        fragment = value
    else:
        fragment = value[:4] + value[-4:]
    return hashlib.sha256(fragment.encode()).hexdigest()[:16]


def _risk_for_provider(provider: str) -> str:
    """Default risk level by provider."""
    return {"aws": "critical", "gcp": "high", "azure": "critical",
            "github": "critical", "stripe": "critical"}.get(provider, "high")


# Entropy-based detection: look for high-entropy strings near key-like variable names
_KEY_CONTEXT_PATTERN = None


def _check_entropy(line: str, line_number: int, file_path: str) -> list[SecretFinding]:
    """Detect high-entropy strings near key/secret/token/password context."""
    import re
    findings = []
    # Find quoted strings near key-like words
    pattern = re.compile(
        r"""(?i)(?:key|secret|token|password|credential|auth)\s*[:=]\s*['"]([^'"]{16,})['"]"""
    )
    for match in pattern.finditer(line):
        value = match.group(1)
        entropy = _shannon_entropy(value)
        if entropy > 4.5:
            findings.append(SecretFinding(
                rule_id="entropy-high",
                provider="generic",
                type="api_key",
                file_path=file_path,
                line_number=line_number,
                fingerprint=_fingerprint(value),
                risk_level="high",
                description=f"High-entropy string (entropy={entropy:.1f}) near sensitive keyword",
            ))
    return findings


def scan_content(content: str, file_path: str) -> list[SecretFinding]:
    """Scan text content for secrets using regex rules + entropy analysis."""
    findings: list[SecretFinding] = []
    lines = content.split("\n")

    for line_idx, line in enumerate(lines, start=1):
        # Layer 1: Regex rules
        for rule in RULES:
            if rule.pattern.search(line):
                # Extract the matched value for fingerprinting
                match = rule.pattern.search(line)
                value = match.group(0) if match else ""
                findings.append(SecretFinding(
                    rule_id=rule.id,
                    provider=rule.provider,
                    type=rule.type,
                    file_path=file_path,
                    line_number=line_idx,
                    fingerprint=_fingerprint(value),
                    risk_level=_risk_for_provider(rule.provider),
                    description=rule.description,
                ))

        # Layer 2: Entropy analysis (only if no regex match on this line)
        if not any(f.line_number == line_idx for f in findings):
            findings.extend(_check_entropy(line, line_idx, file_path))

    return findings
```

```python
"""forge_scan MCP tool — secret scanning."""
import json
import os
from pathlib import Path

from forge_graph.meta import ToolMeta
from forge_graph.security.scanner import scan_content, SecretFinding
from forge_graph.server import mcp, get_db

# File extensions to scan
_SCANNABLE = frozenset({
    ".py", ".js", ".ts", ".tsx", ".jsx", ".go", ".rs", ".java",
    ".yml", ".yaml", ".json", ".toml", ".ini", ".cfg", ".conf",
    ".env", ".sh", ".bash", ".zsh", ".tf", ".tfvars",
    ".xml", ".properties", ".gradle",
})

# Directories to skip
_SKIP_DIRS = frozenset({
    ".git", "node_modules", "__pycache__", ".venv", "venv",
    ".mypy_cache", ".ruff_cache", ".pytest_cache", "dist", "build",
    ".axon", ".lbdb",
})


def _scan_directory(path: Path, depth: str = "shallow") -> list[SecretFinding]:
    """Walk a directory and scan files for secrets."""
    findings: list[SecretFinding] = []
    max_file_size = 1_000_000  # 1MB

    for root, dirs, files in os.walk(path):
        dirs[:] = [d for d in dirs if d not in _SKIP_DIRS]
        for fname in files:
            fpath = Path(root) / fname
            if fpath.suffix not in _SCANNABLE and fpath.name not in {".env", ".npmrc", ".pypirc"}:
                continue
            if fpath.stat().st_size > max_file_size:
                continue
            try:
                content = fpath.read_text(errors="ignore")
                rel_path = str(fpath.relative_to(path))
                findings.extend(scan_content(content, rel_path))
            except (PermissionError, OSError):
                continue

    return findings


@mcp.tool()
async def forge_scan(
    path: str | None = None,
    depth: str = "shallow",
    agent_id: str | None = None,
) -> str:
    """Scan directory for exposed secrets using regex + entropy analysis.

    Returns findings with type, provider, location, and risk level.
    NEVER stores or returns actual secret values — fingerprints only.
    """
    meta = ToolMeta()
    scan_path = Path(path) if path else Path.cwd()

    findings = _scan_directory(scan_path, depth)

    # Store findings as Secret nodes in graph
    db = get_db()
    for f in findings:
        db.conn.execute(
            """MERGE (s:Secret {file_path: $fp, line_number: $ln})
               SET s.type = $type, s.provider = $prov,
                   s.discovered_at = timestamp(), s.risk_level = $risk,
                   s.status = 'active', s.fingerprint = $fp2""",
            parameters={
                "fp": f.file_path, "ln": f.line_number,
                "type": f.type, "prov": f.provider,
                "risk": f.risk_level, "fp2": f.fingerprint,
            },
        )

    return json.dumps({
        "total": len(findings),
        "by_risk": {
            "critical": sum(1 for f in findings if f.risk_level == "critical"),
            "high": sum(1 for f in findings if f.risk_level == "high"),
            "medium": sum(1 for f in findings if f.risk_level == "medium"),
            "low": sum(1 for f in findings if f.risk_level == "low"),
        },
        "findings": [
            {
                "rule": f.rule_id, "provider": f.provider, "type": f.type,
                "file": f.file_path, "line": f.line_number,
                "risk": f.risk_level, "description": f.description,
            }
            for f in findings
        ],
        "_meta": meta.finish(),
    })
```

```python
import re

# Memory node labels that axon_cypher MUST NOT access
_MEMORY_LABELS = frozenset({
    "decision", "pattern", "lesson", "preference", "session", "skill", "secret",
    "_forge_meta",
})

# Write keywords that axon_cypher MUST NOT use

    r"\b(CREATE|SET|DELETE|MERGE|REMOVE|DETACH)\b", re.IGNORECASE
)


def validate_cypher_query(query: str) -> bool:
    """Validate a Cypher query for the axon_cypher sandbox.

    Returns True if the query is safe (read-only, code nodes only).
    Returns False if it accesses memory nodes or attempts writes.
    """
    # Block write operations
    if _WRITE_KEYWORDS.search(query):
        return False

    # Block memory node label access (case-insensitive)
    query_lower = query.lower()
    for label in _MEMORY_LABELS:
        # Match `:Label` pattern (with or without surrounding parens)
        if re.search(rf":{label}\b", query_lower):
            return False

    return True
```

```python
# forge-graph/src/forge_graph/hud/state.py
"""Atomic, debounced HUD state file writer."""
import json
import os
import tempfile
import time
from pathlib import Path


class HudStateWriter:
    """Writes hud-state.json atomically with debouncing and secure permissions."""

    def __init__(self, path: str | Path, debounce_ms: int = 100) -> None:
        self._path = Path(path)
        self._path.parent.mkdir(parents=True, exist_ok=True)
        self._debounce_s = debounce_ms / 1000.0
        self._last_write: float = 0.0
        self._state: dict = {
            "graph": {"nodes": 0, "edges": 0},
            "memory": {"decisions": 0, "patterns": 0, "lessons": 0, "secrets": 0},
            "session": {"mode": None, "phase": None, "wave": None},
            "tokens": {"input": 0, "output": 0, "llm_calls": 0, "deterministic_ratio": 1.0},
            "skills": {"active": 0, "fix_candidates": 0},
            "team": {},
            "security": {"total": 0, "stale": 0, "exposed": 0},
        }
        self._dirty = False

    def update(self, **sections: dict) -> None:
        """Merge section updates into state. Marks dirty."""
        for key, value in sections.items():
            if key in self._state:
                if isinstance(self._state[key], dict) and isinstance(value, dict):
                    self._state[key].update(value)
                else:
                    self._state[key] = value
        self._dirty = True

    def maybe_flush(self) -> None:
        """Flush if dirty and debounce interval has elapsed."""
        if not self._dirty:
            return
        now = time.monotonic()
        if now - self._last_write < self._debounce_s:
            return
        self.flush()

    def flush(self) -> None:
        """Write state to file atomically with secure permissions."""
        if not self._dirty and self._path.exists():
            return

        # Ensure paths in agent state are relative (security)
        for agent_info in self._state.get("team", {}).values():
            if isinstance(agent_info, dict) and "current_file" in agent_info:
                cf = agent_info["current_file"]
                if cf and cf.startswith("/"):
                    agent_info["current_file"] = os.path.basename(cf)

        # Atomic write via temp file + rename
        parent = self._path.parent
        fd, tmp_path = tempfile.mkstemp(dir=parent, suffix=".tmp")
        try:
            with os.fdopen(fd, "w") as f:
                json.dump(self._state, f, separators=(",", ":"))
            os.chmod(tmp_path, 0o600)
            os.rename(tmp_path, self._path)
        except Exception:
            try:
                os.unlink(tmp_path)
            except OSError:
                pass
            raise

        self._last_write = time.monotonic()
        self._dirty = False
```

```rust
        if state.skills.fix_candidates > 0 {

        } else {
            String::new()
        },
    );

    if width >= 120 {
        format!("{tokens_seg} {DIM}|{RESET} {skills_seg} {DIM}|{RESET} {team_seg}")
    } else {

fn render_team(team: &std::collections::HashMap<String, crate::state::AgentInfo>) -> String {
    if team.is_empty() {
        return format!("{DIM}no team{RESET}");
    }
    let mut parts = Vec::new();
    for (name, info) in team {
        let short_name = name.strip_prefix("forge-").unwrap_or(name);
        let icon = match info.status.as_deref() {
            Some("done") => format!("{GREEN}v{RESET}"),
            Some("running") => format!("{YELLOW}*{RESET}"),
            Some("pending") => format!("{DIM}~{RESET}"),
            _ => format!("{DIM}?{RESET}"),
        };
        parts.push(format!("{short_name}{icon}"));
    }
    parts.join(" ")
}
```

```python
"""Evolution safety — output sanitization, path validation, diff limits."""
import re

# Max lines in an evolution diff
MAX_DIFF_LINES = 500

# Allowed path prefixes for evolution writes
ALLOWED_PREFIXES = ("skills/",)

# Blocked paths that must never be modified by evolution
BLOCKED_PATHS = frozenset({
    "hooks/", "scripts/", "agents/", ".claude-plugin/",
    "plugin.json", "hooks.json",
})


def validate_evolution_path(path: str) -> bool:
    """Check if a file path is allowed for evolution writes."""
    # Reject path traversal
    if ".." in path:
        return False

    # Must start with an allowed prefix
    if not any(path.startswith(p) for p in ALLOWED_PREFIXES):
        return False

    # Must not be in blocked paths
    if any(path.startswith(b) for b in BLOCKED_PATHS):
        return False

    return True


def validate_diff_size(diff: str) -> bool:
    """Check if a diff is within size limits."""
    return diff.count("\n") <= MAX_DIFF_LINES


def sanitize_llm_output(output: str) -> str:
    """Strip <think> blocks and other LLM artifacts from output.

    Uses a structured approach — handles nested tags.
    """
    # Remove <think>...</think> blocks (greedy, handles nesting)
    result = output
    while True:
        cleaned = re.sub(r"<think>.*?</think>", "", result, flags=re.DOTALL)
        if cleaned == result:
            break
        result = cleaned

    # Remove any remaining unclosed <think> tags
    result = re.sub(r"<think>[^<]*$", "", result, flags=re.DOTALL)

    return result.strip()
```

```python
# forge-graph/src/forge_graph/evolution/metrics.py
"""Skill metric aggregation from the knowledge graph."""
from forge_graph.db import GraphDB

# Minimum selections before metrics trigger evolution
MIN_SUPPORT = 5

# Sample window
MAX_SESSIONS = 10


def compute_skill_metrics(db: GraphDB, skill_id: str) -> dict:
    """Compute evolution metrics for a skill from graph data."""
    result = db.conn.execute(
        """MATCH (s:Skill {id: $sid})
           RETURN s.total_selections AS sel, s.total_applied AS app,
                  s.total_completions AS comp, s.total_fallbacks AS fb""",
        parameters={"sid": skill_id},
    )
    rows = result.get_as_pl()
    if rows.is_empty():
        return {"cold_start": True, "total_selections": 0}

    sel = int(rows["sel"][0] or 0)
    app = int(rows["app"][0] or 0)
    comp = int(rows["comp"][0] or 0)
    fb = int(rows["fb"][0] or 0)

    if sel < MIN_SUPPORT:
        return {
            "cold_start": True,
            "total_selections": sel,
            "applied_rate": app / sel if sel > 0 else 0,
            "completion_rate": comp / app if app > 0 else 0,
            "effective_rate": comp / sel if sel > 0 else 0,
            "fallback_rate": fb / app if app > 0 else 0,
        }

    return {
        "cold_start": False,
        "total_selections": sel,
        "applied_rate": app / sel if sel > 0 else 0,
        "completion_rate": comp / app if app > 0 else 0,
        "effective_rate": comp / sel if sel > 0 else 0,
        "fallback_rate": fb / app if app > 0 else 0,
    }


def check_evolution_candidates(db: GraphDB) -> list[dict]:
    """Scan all active skills for evolution triggers.

    Returns list of {skill_id, skill_name, evolution_type, reason}.
    Only returns candidates past cold-start guard (>=5 selections).
    """
    candidates = []

    result = db.conn.execute(
        "MATCH (s:Skill) WHERE s.is_active = true RETURN s.id AS id, s.name AS name"
    )
    rows = result.get_as_pl()

    for i in range(len(rows)):
        skill_id = rows["id"][i]
        skill_name = rows["name"][i]
        metrics = compute_skill_metrics(db, skill_id)

        if metrics.get("cold_start", True):
            continue

        # AUTO-FIX: fallback_rate > 0.40
        if metrics["fallback_rate"] > 0.40:
            candidates.append({
                "skill_id": skill_id,
                "skill_name": skill_name,
                "evolution_type": "fix",
                "reason": f"fallback_rate={metrics['fallback_rate']:.2f} > 0.40",
                "metrics": metrics,
            })
        # AUTO-IMPROVE: effective_rate < 0.55 AND applied_rate > 0.25
        elif metrics["effective_rate"] < 0.55 and metrics["applied_rate"] > 0.25:
            candidates.append({
                "skill_id": skill_id,
                "skill_name": skill_name,
                "evolution_type": "improve",
                "reason": f"effective_rate={metrics['effective_rate']:.2f} < 0.55",
                "metrics": metrics,
            })

    return candidates
```

```python
# forge-graph/src/forge_graph/evolution/worktree.py
"""Git worktree lifecycle for safe evolution."""
import os
import shutil
import subprocess
import tempfile
from pathlib import Path


class EvolutionWorktree:
    """Manages an isolated git worktree for evolution writes."""

    def __init__(self, repo_path: str | Path) -> None:
        self._repo = Path(repo_path)
        self._worktree_path: Path | None = None
        self._branch_name: str | None = None

    def create(self) -> Path:
        """Create a temporary worktree for evolution."""
        self._branch_name = f"forge-evolve-{os.urandom(4).hex()}"
        self._worktree_path = Path(tempfile.mkdtemp(prefix="forge-evolve-"))

        subprocess.run(
            ["git", "worktree", "add", "-b", self._branch_name,
             str(self._worktree_path), "HEAD"],
            cwd=self._repo, check=True, capture_output=True,
        )
        return self._worktree_path

    def get_diff(self) -> str:
        """Get the diff of changes in the worktree."""
        if self._worktree_path is None:
            raise RuntimeError("Worktree not created")
        result = subprocess.run(
            ["git", "diff", "--no-color"],
            cwd=self._worktree_path, capture_output=True, text=True,
        )
        return result.stdout

    def apply_to_main(self) -> bool:
        """Apply worktree changes to the main working tree as a patch."""
        if self._worktree_path is None:
            raise RuntimeError("Worktree not created")

        diff = self.get_diff()
        if not diff.strip():
            return False

        result = subprocess.run(
            ["git", "apply", "--check"],
            input=diff, cwd=self._repo, capture_output=True, text=True,
        )
        if result.returncode != 0:
            return False

        subprocess.run(
            ["git", "apply"],
            input=diff, cwd=self._repo, check=True, capture_output=True, text=True,
        )
        return True

    def cleanup(self) -> None:
        """Remove the worktree and temporary branch."""
        if self._worktree_path and self._worktree_path.exists():
            subprocess.run(
                ["git", "worktree", "remove", "--force", str(self._worktree_path)],
                cwd=self._repo, capture_output=True,
            )
        if self._branch_name:
            subprocess.run(
                ["git", "branch", "-D", self._branch_name],
                cwd=self._repo, capture_output=True,
            )
        if self._worktree_path and self._worktree_path.exists():
            shutil.rmtree(self._worktree_path, ignore_errors=True)

    def __enter__(self):
        self.create()
        return self

    def __exit__(self, *args):
        self.cleanup()
```

```python
# forge-graph/src/forge_graph/evolution/engine.py
"""Evolution engine — AUTO-FIX, AUTO-IMPROVE, AUTO-LEARN orchestration.

Uses the autoresearch pattern: modify → measure → keep/discard with git as state machine.
All evolution runs in isolated worktrees. User approves every mutation.
"""
import json
from pathlib import Path

from forge_graph.db import GraphDB
from forge_graph.evolution.metrics import check_evolution_candidates, compute_skill_metrics
from forge_graph.evolution.safety import (
    sanitize_llm_output,
    validate_diff_size,
    validate_evolution_path,
)
from forge_graph.evolution.worktree import EvolutionWorktree


def check_for_suggestions(db: GraphDB) -> list[dict]:
    """Called at SessionStart — returns evolution suggestions for user.

    Pure deterministic (Cypher queries). Zero LLM calls.
    Returns formatted suggestions ready for display.
    """
    candidates = check_evolution_candidates(db)
    suggestions = []

    for c in candidates:
        suggestions.append({
            "skill": c["skill_name"],
            "type": c["evolution_type"],
            "reason": c["reason"],
            "metrics": {
                k: round(v, 2) if isinstance(v, float) else v
                for k, v in c["metrics"].items()
            },
        })

    return suggestions


def format_suggestions_for_user(suggestions: list[dict]) -> str:
    """Format evolution suggestions as user-facing options."""
    if not suggestions:
        return ""

    lines = ["Forge noticed skill health issues:\n"]
    for i, s in enumerate(suggestions, 1):
        action = "AUTO-FIX" if s["type"] == "fix" else "AUTO-IMPROVE"
        lines.append(f"  {i}. {s['skill']}: {s['reason']}")
        lines.append(f"     Suggest: {action}\n")

    lines.append("Options:")
    for i in range(len(suggestions)):
        lines.append(f"  ({chr(97 + i)}) Fix #{i + 1}")
    lines.append(f"  ({chr(97 + len(suggestions))}) Dismiss")
    lines.append(f"  ({chr(98 + len(suggestions))}) Show metrics")

    return "\n".join(lines)


def execute_evolution(
    db: GraphDB,
    skill_id: str,
    evolution_type: str,
    repo_path: str | Path,
) -> dict:
    """Execute a single evolution attempt in an isolated worktree.

    Returns {success, diff, tokens_used, error} for user review.
    This function does NOT apply the changes — it generates the proposal.
    """
    repo = Path(repo_path)

    # 1. Gather evidence (deterministic)
    metrics = compute_skill_metrics(db, skill_id)
    skill_result = db.conn.execute(
        "MATCH (s:Skill {id: $sid}) RETURN s.name AS name",
        parameters={"sid": skill_id},
    )
    skill_rows = skill_result.get_as_pl()
    if skill_rows.is_empty():
        return {"success": False, "error": "Skill not found"}

    skill_name = skill_rows["name"][0]
    skill_path = repo / "skills" / skill_name / "SKILL.md"

    if not skill_path.exists():
        return {"success": False, "error": f"Skill file not found: {skill_path}"}

    current_content = skill_path.read_text()

    # 2. Create worktree (containment)
    try:
        with EvolutionWorktree(repo) as wt:
            wt_skill_path = wt._worktree_path / "skills" / skill_name / "SKILL.md"

            # 3. LLM generates fix (would be called here)
            # For now, return the evidence package for the caller to send to LLM
            return {
                "success": True,
                "needs_llm": True,
                "evidence": {
                    "skill_name": skill_name,
                    "skill_path": str(skill_path.relative_to(repo)),
                    "current_content": current_content,
                    "metrics": metrics,
                    "evolution_type": evolution_type,
                    "worktree_path": str(wt._worktree_path),
                },
            }
    except Exception as e:
        return {"success": False, "error": str(e)}


def validate_and_apply(
    repo_path: str | Path,
    worktree_path: str | Path,
    llm_output: str,
    target_path: str,
) -> dict:
    """Validate LLM output and apply evolution if safe.

    Called after user approves the proposed diff.
    """
    # 1. Validate target path
    if not validate_evolution_path(target_path):
        return {"success": False, "error": f"Path not allowed: {target_path}"}

    # 2. Sanitize LLM output
    clean = sanitize_llm_output(llm_output)

    # 3. Write to worktree
    wt_path = Path(worktree_path) / target_path
    wt_path.parent.mkdir(parents=True, exist_ok=True)
    wt_path.write_text(clean)

    # 4. Get diff and validate size
    wt = EvolutionWorktree(repo_path)
    wt._worktree_path = Path(worktree_path)
    diff = wt.get_diff()

    if not validate_diff_size(diff):
        return {"success": False, "error": f"Diff too large ({diff.count(chr(10))} lines > 500 max)"}

    # 5. Apply patch to main tree
    applied = wt.apply_to_main()
    if not applied:
        return {"success": False, "error": "Patch failed to apply cleanly"}

    return {"success": True, "diff": diff, "lines_changed": diff.count("\n")}
```

```python
def _decisions_impl(
    db: GraphDB,
    code_path: str | None = None,
    symbol: str | None = None,
    agent_id: str | None = None,
) -> str:
    """What decisions affect this code path or symbol?"""
    meta = ToolMeta()
    results = []

    if code_path:
        # Query across all AFFECTS edge variants
        for edge, target in [("AFFECTS", "Function"), ("AFFECTS_CLASS", "Class"), ("AFFECTS_FILE", "File")]:
            result = db.conn.execute(
                f"""MATCH (d:Decision)-[:{edge}]->(t:{target})
                    WHERE t.file_path CONTAINS $path AND d.invalid_at IS NULL
                    RETURN d.id AS id, d.title AS title, d.rationale AS rationale,
                           d.status AS status, t.name AS target_name, t.file_path AS target_path""",
                parameters={"path": code_path},
            )
            rows = result.get_as_pl()
            for i in range(len(rows)):
                results.append({
                    "id": rows["id"][i], "title": rows["title"][i],
                    "rationale": rows["rationale"][i], "status": rows["status"][i],
                    "affects": rows["target_name"][i], "file": rows["target_path"][i],
                })

    if symbol:
        result = db.conn.execute(
            """MATCH (d:Decision)-[:AFFECTS]->(f:Function {name: $sym})
               WHERE d.invalid_at IS NULL
               RETURN d.id AS id, d.title AS title, d.rationale AS rationale,
                      d.status AS status, f.name AS target_name, f.file_path AS target_path""",
            parameters={"sym": symbol},
        )
        rows = result.get_as_pl()
        for i in range(len(rows)):
            results.append({
                "id": rows["id"][i], "title": rows["title"][i],
                "rationale": rows["rationale"][i], "status": rows["status"][i],
                "affects": rows["target_name"][i], "file": rows["target_path"][i],
            })

    return json.dumps({"results": results, "_meta": meta.finish()})


def _timeline_impl(
    db: GraphDB,
    node_id: str,
    node_label: str = "Decision",
    depth: int = 10,
    agent_id: str | None = None,
) -> str:
    """Follow SUPERSEDES/EVOLVED_FROM chains for temporal evolution."""
    meta = ToolMeta()
    chain = []

    if node_label == "Decision":
        result = db.conn.execute(
            """MATCH path = (start:Decision {id: $nid})-[:SUPERSEDES*1..10]->(older:Decision)
               RETURN older.id AS id, older.title AS title, older.status AS status""",
            parameters={"nid": node_id},
        )
    elif node_label == "Skill":
        result = db.conn.execute(
            """MATCH path = (start:Skill {id: $nid})-[:EVOLVED_FROM*1..10]->(older:Skill)
               RETURN older.id AS id, older.name AS title, older.generation AS status""",
            parameters={"nid": node_id},
        )
    else:
        return json.dumps({"chain": [], "_meta": meta.finish()})

    rows = result.get_as_pl()
    for i in range(len(rows)):
        chain.append({
            "id": rows["id"][i],
            "title": rows["title"][i],
            "status": str(rows["status"][i]),
        })

    return json.dumps({"chain": chain, "_meta": meta.finish()})


@mcp.tool()
async def forge_decisions(
    code_path: str | None = None,
    symbol: str | None = None,
    agent_id: str | None = None,
) -> str:
    """Show all decisions that affect a code path or symbol. The killer query."""
    return _decisions_impl(get_db(), code_path, symbol, agent_id)


@mcp.tool()
async def forge_timeline(
    node_id: str,
    node_label: str = "Decision",
    depth: int = 10,
    agent_id: str | None = None,
) -> str:
    """Follow temporal evolution chains — SUPERSEDES for decisions, EVOLVED_FROM for skills."""
    return _timeline_impl(get_db(), node_id, node_label, depth, agent_id)
```

```bash
#!/usr/bin/env bash
# scripts/post-edit-enhanced.sh — Decision awareness + secret detection on edit
set -euo pipefail

SCRIPT_DIR="$(cd "$(dirname "$(readlink -f "$0")")" && pwd)"
PLUGIN_ROOT="$(dirname "$SCRIPT_DIR")"

# Read tool input from stdin
INPUT=$(cat)
FILE_PATH=$(echo "$INPUT" | jq -r '.toolInput.file_path // .toolInput.filePath // empty' 2>/dev/null)

if [ -z "$FILE_PATH" ]; then
    exit 0
fi

# Resolve canonical path (symlink defense)
CANONICAL=$(readlink -f "$FILE_PATH" 2>/dev/null || echo "$FILE_PATH")
WORKSPACE=$(readlink -f "$(pwd)")

# Workspace boundary check
if [[ "$CANONICAL" != "$WORKSPACE"* ]]; then
    echo '{"decision":"block","reason":"Path outside workspace boundary"}'
    exit 0
fi

# Quick secret scan (Layer 1 — regex only, fast)
ALERTS=""
if [ -f "$CANONICAL" ]; then
    # AWS keys
    if grep -qP 'AKIA[A-Z0-9]{16}' "$CANONICAL" 2>/dev/null; then
        ALERTS="${ALERTS}AWS access key detected. "
    fi
    # GitHub PATs
    if grep -qP 'ghp_[A-Za-z0-9]{36,}' "$CANONICAL" 2>/dev/null; then
        ALERTS="${ALERTS}GitHub PAT detected. "
    fi
    # Private keys
    if grep -q 'BEGIN.*PRIVATE KEY' "$CANONICAL" 2>/dev/null; then
        ALERTS="${ALERTS}Private key detected. "
    fi
    # High-entropy near sensitive keywords
    if grep -qPi '(password|secret|token|api_key)\s*[:=]\s*["\x27][^\s"'\'']{16,}' "$CANONICAL" 2>/dev/null; then
        ALERTS="${ALERTS}Possible hardcoded secret. "
    fi
fi

if [ -n "$ALERTS" ]; then
    echo "{\"hookSpecificOutput\":{\"additionalContext\":\"SECRET ALERT in ${FILE_PATH}: ${ALERTS}Consider moving to .env or .gitignore.\"}}"
fi

# Existing: run formatter (from post-edit-format.sh)
```
